def print(node, io: StringIO.new)
  lines = pretty_output_lines(node)

  lines.each do |line|
    if line.text.empty?
      io << line.sexpr << "\n"
      next
    end

    io << "#{line.sexpr}#{" " * (@ralign - line.sexpr.size)}| #{line.text}\n"
  end

  io
end
